# 테스트 코드 예시

```go
package mypackage

import (
        "testing"
)

func TestMyFunction(t *testing.T) {

        got, err := MyFunction()
        assert.NoError(t, err)
        assert.Equal(t, "Hello", got)

}

func TestService_MyHttpFunction(t *testing.T) {


        jar, _ := cookiejar.New(nil)
        client := &http.Client{
                Jar: jar,
        }

        svc := New()

        s := `<HTML></HTML>`
        ts := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
                fmt.Fprintln(w, s)
        }))
        defer ts.Close()

        svc.MyHttpFunctionUrl = ts.URL

        got, err := svc.MyHttpFunction(client)

        assert.NoError(t, err)
        assert.Equal(t, "Hello", got)

}
```
