# 테스트 코드 예시

```go
package mypackage

import (
        "testing"
)

func TestMyFunction(t *testing.T) {

        got, err := MyFunction()
        assert.NoError(t, err)
        assert.Equal(t, "Hello", got)

}

type dummyLog struct{}

func (m *dummyLog) Println(a ...any) (n int, err error) {
        return fmt.Println(a...)
}

func (m *dummyLog) Printf(format string, a ...any) (n int, err error) {
        return fmt.Printf(format, a...)
}

func newHttpClient() *http.Client {
        jar, _ := cookiejar.New(&cookiejar.Options{PublicSuffixList: publicsuffix.List})
        client := &http.Client{
                Jar: jar,
        }
        return client
}

func newHttpTestServer() *httptest.Server {

        s := `<HTML></HTML>`
        ts := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
                fmt.Fprintln(w, s)
        }))
        return ts
}

func TestService_MyHttpFunction(t *testing.T) {


        client := newHttpClient()
        ts := newHttpTestServer()
        defer ts.Close()

        svc := New(&dummyLog{}, func() time.Duration { return 0 })

        svc.MyHttpFunctionUrl = ts.URL
        got, err := svc.MyHttpFunction(client)

        assert.NoError(t, err)
        assert.Equal(t, "Hello", got)

}
```
