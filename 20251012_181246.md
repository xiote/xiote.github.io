# 테스트 코드 템플릿

```go
package mypackage

import (
        "testing"

        "github.com/stretchr/testify/assert"
)

func MyFunction() (string, error) {
        return "Hello, Alice", nil
}

func TestMyFunction(t *testing.T) {

        got, err := MyFunction()
        assert.NoError(t, err)
        assert.Equal(t, "Hello, Alice", got)

}
```
